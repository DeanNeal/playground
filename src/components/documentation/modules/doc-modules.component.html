Modules

To simplify app.js you can split your code into modules
Call your module as index.js for easier using, so then just import components/documentation where index.js file is.
index.js
        import {DocumentationComponent} from './documentation.component';
        import {DocQuickStartComponent} from './quick-start/doc-quick-start.component';

        export default [
            {c: DocumentationComponent, selector: 'app-documentation'},
            {c: DocQuickStartComponent, selector: 'app-documentation-quick-start'}
            ...
        ]
app.js
        import Docs from './components/documentation';
        Register({
            ...
            modules: [
                Docs
                ...
            ]
            ...
        });
